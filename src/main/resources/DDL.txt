


CREATE TABLE USER_TB
(
		USER_NO_PK INT NOT NULL AUTO_INCREMENT,
		USER_ID VARCHAR(200),
		USER_PW VARCHAR(500),
		USER_EMAIL VARCHAR(500),
PRIMARY KEY(USER_NO_PK)
) ENGINE=MYISAM CHARSET=utf8;


CREATE TABLE FILE_TB
(
	POST_NO INT NOT NULL,
	FILE_SN INT,
	ORIGINAL_NAME VARCHAR(500),
	STORED_NAME VARCHAR(500),
	FILE_SIZE INT

) ENGINE=MYISAM CHARSET=utf8;




CREATE TABLE POST_TB
(
  		POST_NO_PK INT NOT NULL AUTO_INCREMENT,
		TITLE		VARCHAR(500),
		CONTENT		VARCHAR(4000),
		WRT_NO		INT,
		WRT_DATE	DATE,
		WRT_RE_DATE DATE,
		VIEW_CNT	INT,
		DEL_FL		CHAR(1),
PRIMARY KEY(POST_NO_PK)
) ENGINE=MYISAM CHARSET=utf8;
